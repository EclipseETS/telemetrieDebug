char * DataReceived;

NRF_SET_PRX();	
NRF_Listen();

if (!(IRQ_PORT_READ & (1<<IRQ_PIN)))											
{
	Status = NRF_ReadStatusRegister();											
	if (Status & (1<<RX_DR))													
	{												
		DataReceived = NRF_ReadRXPayload(2);		// Read data from FIFO.																	
		NRF_FlushFifo(RX);														
		NRF_WriteRegister(STATUS, (1<<RX_DR));		// Reset register.	
	}
}
